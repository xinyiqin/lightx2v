cmake_minimum_required(VERSION 3.22 FATAL_ERROR)
project(sycl-kernels LANGUAGES CXX)

find_package(IntelSYCL REQUIRED)

# Python (scikit-build-core injects paths; this finds Interpreter + Development.Module)
find_package(Python COMPONENTS Interpreter Development.Module REQUIRED)

# ── ur_api.h shim ─────────────────────────────────────────────────────────────
# conda puts an old ur_api.h that lacks symbols required by oneAPI 2025.x.
# FindIntelSYCL (called above) already resolves the 'latest' junction and sets
# SYCL_INCLUDE_DIR to the real versioned path (e.g. .../compiler/2025.2/include).
# Use that directly so the shim never embeds the unresolvable 'latest' symlink.
file(MAKE_DIRECTORY "${CMAKE_BINARY_DIR}/includes")
file(WRITE "${CMAKE_BINARY_DIR}/includes/ur_api.h"
    "#pragma once\n// Shim: redirect to the compiler-matched ur_api.h\n// Generated by CMake -- do not edit by hand.\n#include \"${SYCL_INCLUDE_DIR}/ur_api.h\"\n"
)
set(CMAKE_CXX_FLAGS "/I\"${CMAKE_BINARY_DIR}/includes\" ${CMAKE_CXX_FLAGS}")
message(STATUS "ur_api.h shim → ${SYCL_INCLUDE_DIR}/ur_api.h")

find_package(Torch REQUIRED)

# Extension target: _ext (imported as sycl_kernels._ext)
set(MODULE_NAME _ext)

set(SYCL_SRCS csrc/onednn.cpp csrc/onednn_fp8.cpp)
set(ALL_SRCS ${SYCL_SRCS} csrc/entry.cpp csrc/sdp.cpp)

Python_add_library(${MODULE_NAME} MODULE WITH_SOABI ${ALL_SRCS})

target_compile_definitions(${MODULE_NAME} PRIVATE TORCH_EXTENSION_NAME=${MODULE_NAME})

add_sycl_to_target(TARGET ${MODULE_NAME} SOURCES ${SYCL_SRCS})

if(WIN32)
    target_compile_options(${MODULE_NAME} PRIVATE -Wno-ignored-attributes)
endif()

# Force oneAPI compiler headers before conda's conflicting headers.
# SYCL_INCLUDE_DIR is the real versioned path set by FindIntelSYCL.
if(CMAKE_CXX_COMPILER_ID MATCHES "IntelLLVM")
    target_include_directories(${MODULE_NAME} BEFORE PRIVATE
        "${SYCL_INCLUDE_DIR}/sycl"
        "${SYCL_INCLUDE_DIR}"
    )
endif()

target_include_directories(${MODULE_NAME} PRIVATE "${CMAKE_BINARY_DIR}/includes")
target_include_directories(${MODULE_NAME} PRIVATE ${TORCH_INCLUDE_DIRS})

# oneDNN library (from active conda env or oneAPI install)
execute_process(
    COMMAND ${Python_EXECUTABLE} -c "import sys; print(sys.prefix)"
    OUTPUT_VARIABLE Python_PREFIX
    OUTPUT_STRIP_TRAILING_WHITESPACE
)
find_library(DNNL_LIBRARY
    NAMES dnnl
    PATHS
        "${Python_PREFIX}/Library/lib"
        "${Python_PREFIX}/lib"
        "${Python_PREFIX}/lib64"
        "$ENV{ONEAPI_ROOT}/dnnl/latest/lib"
        "C:/Program Files (x86)/Intel/oneAPI/dnnl/2025.2/lib"
        "C:/Program Files/Intel/oneAPI/dnnl/2025.2/lib"
    NO_DEFAULT_PATH
)
if(NOT DNNL_LIBRARY)
    message(FATAL_ERROR
        "oneDNN (dnnl) not found under ${Python_PREFIX}.\n"
        "Activate the lightx2v_kernel conda env before building.")
endif()
message(STATUS "Found oneDNN: ${DNNL_LIBRARY}")

# ESIMD SDP DLL import library (must be built first — Step 1 of build_la.bat)
add_library(esimd_lgrf_lib STATIC IMPORTED)
set_target_properties(esimd_lgrf_lib PROPERTIES
    IMPORTED_LOCATION "${CMAKE_CURRENT_SOURCE_DIR}/lgrf_uni/esimd.unify.lgrf.lib")

find_library(TORCH_PYTHON_LIBRARY torch_python PATH "${TORCH_INSTALL_PREFIX}/lib")

target_link_libraries(${MODULE_NAME} PRIVATE
    ${TORCH_LIBRARIES}
    ${TORCH_PYTHON_LIBRARY}
    ${DNNL_LIBRARY}
    esimd_lgrf_lib
)

# ── Install targets for wheel assembly ────────────────────────────────────────
# scikit-build-core collects install() outputs into the wheel platlib
install(TARGETS ${MODULE_NAME} LIBRARY DESTINATION sycl_kernels)
install(FILES lgrf_uni/esimd.unify.lgrf.dll DESTINATION sycl_kernels)
